引用方式

  	SWIPER官网地址：http://www.swiper.com.cn/demo/index.html

  	指令名称 : ui-swiper 
  	swiper参数 : data-config="{zoom: true,pagination: '.swiper-pagination',nextButton: '.swiper-button-next',prevButton: '.swiper-button-prev'}"
  	回调实体 : complete => function($object){};  $object =>  swiper对象

  	指令源码源码 ：

	/**
	 * 滑动滚动
	 * @complete => "初始完毕后回调"
	 * @config =>  "swiper选项配置" 
	 */
	lwjui.directive('ui-swiper', function() {
		return {
			uses: ["plugin/swiper/js/swiper.js"],
			addcss: ['plugin/swiper/swiper.css'],
			scope: {
				config: '=config',
				callback:'~complete'
			},
			link: function(el) {
				var scope = el.scope,
					service = el.fn.swiper;
				// 扩展回调方法
				var service = service.init(el.element, scope, el.fn.cache.dir + "plugin/swiper/");
				if(scope.callback){
					scope.callback(service)
				}				
			}
		};
	});
